Código de ejemplo


// Inicialización básica
const tabla = new RojeruTableSorterLite('demoTable', {
    data: datosEjemplo,
    columns: [
        { key: 'id', title: 'ID', sortable: true },
        { key: 'nombre', title: 'Nombre', sortable: true },
        { key: 'edad', title: 'Edad', sortable: true },
        { key: 'ciudad', title: 'Ciudad', sortable: true },
        { key: 'estado', title: 'Estado',
          selectOptions: [
            { value: 'activo', label: 'Activo' },
            { value: 'inactivo', label: 'Inactivo' }
          ]
        }
    ],
    rowsPerPage: 10,
    showSearch: true,
    locale: 'es'
});

// Agregar fila
document.getElementById('btnAdd').addEventListener('click', () => {
    tabla.addRow({
        id: tabla.getTotalRecords() + 1,
        nombre: 'Nuevo usuario',
        edad: Math.floor(Math.random() * 50) + 18,
        ciudad: ciudades[Math.floor(Math.random() * ciudades.length)],
        estado: Math.random() > 0.5 ? 'activo' : 'inactivo'
    });
});
